};

};

/// Represents a server node that implements both the NetworkNode and HostNode traits.
/// It is responsible for handling network packets, simulation events, file requests, chat, and more.
pub struct Server {
    id: NodeId,
    server_type: ServerType,
    sim_contr_send: Sender<DroneEvent>,
    sim_contr_recv: Receiver<ServerCommand>,
    packet_recv: Receiver<Packet>,
    packet_send: HashMap<NodeId, Sender<Packet>>,
    seen_flood_ids: HashSet<String>,
    topology: Option<Config>,
    random_generator: StdRng,
    session_fragments: HashMap<u64, Vec<Packet>>, // session_id -> Packet
    pending_fragments: HashMap<(u64, u64, NodeId), Fragment>, // (session_id, fragment_id) -> Fragment
    nack_for_error_in_routing_fragments: HashMap<(u64, u64, NodeId), Fragment>, // (session_id, fragment_id) -> Fragment
    pending_serializable_messages: Vec<(NodeId, SerializableMessage)>,
    registred_clients: Vec<NodeId>,
    pdr_estimation_map: HashMap<NodeId, (i32, i32)>,
}

impl NetworkNode for Server {
    /// Returns the node identifier of the server.
    fn get_id(&self) -> NodeId {
        self.id
    }
    
    /// Returns a mutable reference to the set of flood identifiers that have been seen.
    fn get_seen_flood_ids(&mut self) -> &mut HashSet<String> {
        &mut self.seen_flood_ids
    }
    
    /// Returns a mutable reference to the map of packet senders.
    fn get_packet_send(&mut self) -> &mut HashMap<NodeId, Sender<Packet>> {
        &mut self.packet_send
    }
    
    /// Returns a reference to the packet receiver.
    fn get_packet_receiver(&self) -> &Receiver<Packet> {
        &self.packet_recv
    }
    
    /// Returns a mutable reference to the server's random number generator.
    fn get_random_generator(&mut self) -> &mut StdRng {
        &mut self.random_generator
    }
    
    /// Returns a reference to the simulation controller sender.
    fn get_sim_contr_send(&self) -> &Sender<DroneEvent> {
        &self.sim_contr_send
    }

    /// Handles a routed packet by delegating to the host node's routing packet handler.
    fn handle_routed_packet(&mut self, packet: Packet) -> bool {
        self.host_handle_routed_packet(packet)
    }
    
    /// Handles an incoming network command and processes it based on its type.
    fn handle_command(&mut self, command: Command) {
        match command {
            Command::Server(server_command) => match server_command {
                ServerCommand::RemoveSender(node_id) => self.remove_channel(node_id),
                ServerCommand::AddSender(node_id, sender) => self.add_channel(node_id, sender),
                ServerCommand::SetServerType(server_type) => self.set_server_type(server_type),
            },
            _ => {
                log_error!(self.id, "Received a wrong command type");
                panic!("Server {} received a wrong command type", self.get_id());
            }
        }
    }
}

impl HostNode for Server {
    /// Runs the server's main loop, handling simulation controller events and incoming packets.
    fn run(&mut self) {
        // Flooding: initialize topology and initiate flood request.
        self.initialize_topology(); 
        let _topology_str = self.get_topology_print_string(0);
        // log_status!(self.id, "{}", _topology_str);
        self.initiate_flood_request();
        
        loop {
            select_biased!(
                recv(self.sim_contr_recv) -> command_res => {
                    if let Ok(server_command) = command_res {
                        let command = Command::Server(server_command);
                        self.handle_command(command);
                    }
                },
                recv(self.packet_recv) -> packet_res => {
                    if let Ok(packet) = packet_res {
                        self.handle_packet(packet, NodeType::Server);
                    }
                }
            );
        }
    }
    
    /// Indicates that this node is not a client.
    fn am_i_client(&self) -> bool {
        false
    }
    
    /// Returns the current topology configuration, if available.
    fn get_topology(&self) -> Option<Config> {
        self.topology.clone()
    }
    
    /// Sets the topology configuration for the server.
    fn set_topology(&mut self, topology: Config) {
        self.topology = Some(topology);
    }
    
    /// Retrieves the fragments associated with a given session.
    /// Returns an empty vector if no fragments exist for the session.
    fn get_single_session_fragments(&self, session: u64) -> &Vec<Packet> {
        match self.session_fragments.get(&session) {
            Some(fragments) => fragments,
            None => {
                static EMPTY_VEC: Vec<Packet> = Vec::new();
                &EMPTY_VEC
            }
        }
    }
    
    /// Sets the fragments for a specific session.
    fn set_single_session_fragments(&mut self, session: u64, session_fragments: Vec<Packet>) {
        self.session_fragments.insert(session, session_fragments);
    }
    
    /// Removes all fragments associated with a specific session.
    fn remove_session_fragments(&mut self, session: u64) {
        self.session_fragments.remove(&session);
    }

    /// Returns a clone of the map containing pending fragments.
    fn get_pending_fragments(&self) -> HashMap<(u64, u64, NodeId), Fragment> {
        self.pending_fragments.clone()
    }

    /// Sets the map of pending fragments.
    fn set_pending_fragments(&mut self, pending_fragments: HashMap<(u64, u64, NodeId), Fragment>) {
        self.pending_fragments = pending_fragments;
    }
    
    /// Returns a clone of the fragments that need to be NACKed due to routing errors.
    fn get_nack_for_error_in_routing_fragments(&self) -> HashMap<(u64, u64, NodeId), Fragment> {
        self.nack_for_error_in_routing_fragments.clone()
    }
    
    /// Sets the fragments that should be NACKed due to errors in routing.
    fn set_nack_for_error_in_routing_fragments(&mut self, fragments: HashMap<(u64, u64, NodeId), Fragment>) {
        self.nack_for_error_in_routing_fragments = fragments;
    }    
    
    /// Returns a clone of the pending serializable messages.
    fn get_pending_serializable_messages(&self) -> Vec<(NodeId, SerializableMessage)> {
        self.pending_serializable_messages.clone()
    }
    
    /// Sets the list of pending serializable messages.
    fn set_pending_serializable_messages(&mut self, messages: Vec<(NodeId, SerializableMessage)>) {
        self.pending_serializable_messages = messages;
    }

    /// Retrieves the packet delivery ratio (PDR) estimation for a specific node.
    fn get_pdr_estimation_for_node(&self, node_id: NodeId) -> (i32, i32) {
        *self.pdr_estimation_map.get(&node_id).unwrap_or(&(0, 0))
    }

    /// Sets the packet delivery ratio (PDR) estimation for a specific node.
    fn set_pdr_estimation_for_node(&mut self, node_id: NodeId, pdr: (i32, i32)) {
        self.pdr_estimation_map.insert(node_id, pdr);
    }

    /// Handles a file list request from a client. If the server is a content server, it sends the file list;
    /// otherwise, it sends an error message.
    fn handle_file_list_request(&mut self, client_id: NodeId) {
        
        log_status!(
            self.id,
            "Received file list request from CLIENT {}",
            client_id
        );
        
        // Check if the server is a content server.
        match self.get_server_type() {
            ServerType::Content => {},
            _ => {
                self.build_and_forward_serializable_packets(
                    client_id,
                    SerializableMessage::ErrorMessage(client_id, format!("SERVER {} is not a content server", self.get_id())),
                );
                
                log_error!(
                    self.id,
                    "I am not a content server, sent error to CLIENT {}",
                    client_id,
                );
                return;
            }
        }
        
        self.build_and_forward_serializable_packets(
            client_id,
            SerializableMessage::FileListResponse(
                self.id,
                self.get_file_list()
            )
        );
        
        log_status!(self.id, "Sent file list response to CLIENT {}", client_id);
    }
        
    /// Handles a file request from a client. If the file is found and the server is a content server,
    /// it sends the file; otherwise, it responds with an error.
    fn handle_file_request(&mut self, client_id: NodeId, filename: String) {
        log_status!(
            self.id,
            "Received request for file '{}' from CLIENT {}",
            filename,
            client_id
        );

        match self.get_server_type() {
            ServerType::Content => {},
            _ => {
                self.build_and_forward_serializable_packets(
                    client_id,
                    SerializableMessage::ErrorMessage(
                        client_id,
                        format!("SERVER {} is not a content server", self.get_id())
                    ),
                );

                log_error!(
                    self.id,
                    "I am not a content server, sent error to CLIENT {}",
                    client_id,
                );
                return;
            }
        }
        
        let file_list: Vec<String> = self.get_file_list();
        
        if file_list.contains(&filename) {
            let file = self.get_file(filename.clone());
            match file {
                Ok(file) => {

                    self.build_and_forward_serializable_packets(
                        client_id,
                        SerializableMessage::FileFound(
                            self.id,
                            filename.clone(),
                            file
                        ),
                    );

                    log_status!(self.id, "Sent file '{}' to CLIENT {}", filename, client_id);
                    return;
                }
                Err(e) => {
                    log_error!(self.id, "Error reading file: {}", e);
                }
            }
        }

        self.build_and_forward_serializable_packets(
            client_id,
            SerializableMessage::ErrorMessage(
                    self.id,
                    format!("File '{}' not found on SERVER {}.", filename.clone(), self.get_id())
                ),
        );

        log_status!(
            self.id,
            "File '{}' not found, sent error to CLIENT {}",
            filename,
            client_id
        );
    }
    
    /// Handles a server type request from a client by sending the current server type.
    fn handle_server_type_request(&mut self, client_id: NodeId) {
        log_status!(
            self.id,
            "Received server type request from CLIENT {}",
            client_id
        );

        let server_type = self.get_server_type();

        self.build_and_forward_serializable_packets(
            client_id,
            SerializableMessage::ServerTypeResponse(self.id, server_type.clone())
        );

        log_status!(
            self.id,
            "Sent my server type {:?} to CLIENT {}",
            server_type,
            client_id
        );
    }
    
    /// Handles a registration request from a client for a communication server.
    /// If the server is not a communication server or the client is already registered,
    /// an error is sent; otherwise, the client is registered.
    fn handle_register_to_communication_server(&mut self, client_id: NodeId){
        log_status!(
            self.id,
            "Received communication registration request from CLIENT {}",
            client_id
        );
        
        let server_type = self.get_server_type();
        
        if !matches!(server_type, ServerType::Communication) {
            
            self.build_and_forward_serializable_packets(
                    client_id,
                    SerializableMessage::ErrorMessage(
                        self.get_id(),
                        format!("SERVER {} is not a communication server", self.get_id())
                    ),
            );
            
            log_status!(
                self.id,
                "I am not a content server, sent error to CLIENT {}",
                client_id
            );
            return;
        }
        
        let registred_clients = self.get_registred_clients();

        if registred_clients.contains(&client_id) {
            self.build_and_forward_serializable_packets(
                    client_id,
                    SerializableMessage::ErrorMessage(
                        client_id,
                        format!("CLIENT {} is already registered to SERVER {}", client_id, self.get_id())
                    ),
            );
            
            log_status!(
                self.id,
                "Client {} is already registered to me",
                client_id
            );
            
            return;
        }

        self.add_registred_client(client_id);

        self.build_and_forward_serializable_packets(
            client_id,
            SerializableMessage::RegisterSuccess(self.id),
        );
        
        log_status!(
            self.id,
            "Sent communication server registration success to CLIENT {}",
            client_id
        );
    }
    
    /// Handles a client list request by sending the list of registered clients if the server is a communication server.
    fn handle_client_list_request(&mut self, client_id: NodeId) {

        log_status!(
            self.id,
            "Received client list request from CLIENT {}",
            client_id
        );

        match self.get_server_type() {
            ServerType::Communication => {},
            _ => {

                self.build_and_forward_serializable_packets(
                    client_id,
                    SerializableMessage::ErrorMessage(
                        self.get_id(),
                        format!("SERVER {} is not a communication server", self.get_id())
                    ),
                );

                log_error!(
                    self.id,
                    "I am not a content server, sent error to CLIENT {}",
                    client_id
                );
                return;
            }
        }

        self.build_and_forward_serializable_packets(
            client_id,
            SerializableMessage::ClientListResponse(
                self.id,
                self.get_registred_clients().clone()
            ),
        );
        
        log_status!(self.id, "Sent client list response to CLIENT {}", client_id);
    }

    /// Handles a chat message sent from one client to another.
    /// If both the sender and recipient are registered to the communication server, the message is forwarded;
    /// otherwise, an error message is sent back to the sender.
    fn handle_chat(&mut self, sending_client_id: NodeId, _: NodeId, recipient_client_id: NodeId, message: String) {
        log_status!(
            self.id,
            "Received chat from CLIENT {} for CLIENT {}",
            sending_client_id,
            recipient_client_id
        );

        match self.get_server_type() {
            ServerType::Communication => {},
            _ => {

                self.build_and_forward_serializable_packets(
                    sending_client_id,
                    SerializableMessage::ErrorMessage(
                        self.get_id(),
                        format!("SERVER {} is not a communication server", self.get_id())
                    ),
                );

                log_error!(
                    self.id,
                    "I am not a content server, sent error to CLIENT {}",
                    sending_client_id
                );
                return;
            }
        }

        let is_recipient_registred = self.get_registred_clients().contains(&recipient_client_id);
        let is_sender_registred = self.get_registred_clients().contains(&sending_client_id);

        let error_message = match (is_recipient_registred, is_sender_registred) {
            (true, true) => None,
            (false, true) => Some(format!("CLIENT {} is not registered to SERVER {}", recipient_client_id, self.get_id())),
            (true, false) => Some(format!("CLIENT {} is not registered to SERVER {}", sending_client_id, self.get_id())),
            (false, false) => Some(format!("CLIENT {} and CLIENT {} are not registered to SERVER {}", sending_client_id, recipient_client_id, self.get_id())),
        };

        if error_message.is_none() {
            self.build_and_forward_serializable_packets(
                recipient_client_id,
                SerializableMessage::Chat(
                    sending_client_id, 
                    self.id, 
                    recipient_client_id, 
                    message
                ),
            );
            log_status!(self.id, "Sent chat message to CLIENT {}", recipient_client_id);
            return;
        }

        self.build_and_forward_serializable_packets(
            sending_client_id,
            SerializableMessage::ErrorMessage(self.get_id(), error_message.expect("Error message is None")),
        );

        log_error!(
            self.id,
            "CLIENT {} and/or CLIENT {} is not registered to me, sent error to CLIENT {}",
            sending_client_id,
            recipient_client_id,
            sending_client_id
        );
    }
}

impl Server {
    /// Creates a new server instance with the specified parameters.
    pub fn new(
        id: NodeId,
        controller_send: Sender<DroneEvent>,
        controller_recv: Receiver<ServerCommand>,
        packet_recv: Receiver<Packet>,
        packet_send: HashMap<NodeId, Sender<Packet>>,
    ) -> Self {
        Self {
            id,
            server_type: ServerType::Undefined,
            sim_contr_send: controller_send,
            sim_contr_recv: controller_recv,
            packet_recv,
            packet_send,
            seen_flood_ids: HashSet::new(),
            topology: None,
            random_generator: StdRng::from_entropy(),
            session_fragments: HashMap::new(),
            pending_fragments: HashMap::new(),
            nack_for_error_in_routing_fragments: HashMap::new(),
            pending_serializable_messages: Vec::new(),
            registred_clients: Vec::<NodeId>::new(),
            pdr_estimation_map: HashMap::new(),
        }
    }
    
    /// Returns the current server type.
    fn get_server_type(&self) -> ServerType {
        self.server_type.clone()
    }
    
    /// Sets the server type and logs the change.
    fn set_server_type(&mut self, server_type: ServerType) {
        log_status!(self.id, "Server type changed to {:?}", server_type);
        self.server_type = server_type;
    }
    
    /// Retrieves the list of files available in the server's content directory.
    fn get_file_list(&self) -> Vec<String> {
        let folder_name = format!("content/{}", self.get_id());
        let dir_path = Path::new(&folder_name);
        
        let entries = fs::read_dir(dir_path);
        return match entries {
            Ok(entries) => entries
            // Filter out erroneous entries and convert file names to strings.
            .filter_map(|entry| entry.ok())
            .map(|entry| entry.file_name().into_string().expect("Failed to convert file name into string"))
            .collect(),
            Err(_) => Vec::new(),
        };
    }
    
    /// Reads the specified file from the server's content directory and returns its base64-encoded contents.
    fn get_file(&self, filename: String) -> Result<String, io::Error> {
        let path = format!("content/{}/{}", self.id, filename);
        let file_data = fs::read(path)?;
        let base64_encoded = base64::encode(file_data);
        
        Ok(base64_encoded)
    }
    
    /// Returns a reference to the list of registered clients.
    fn get_registred_clients(&self) -> &Vec<NodeId> {
        &self.registred_clients
    }
    
    /// Adds a client to the list of registered clients.
    fn add_registred_client(&mut self, client_id: NodeId) {
        self.registred_clients.push(client_id);
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    
    /// Tests that a new server is created with the expected initial state.
    #[test]
    fn test_server_creation() {
        let (controller_send, _) = crossbeam_channel::unbounded();
        let (_, controller_recv) = crossbeam_channel::unbounded();
        let (_, packet_recv) = crossbeam_channel::unbounded();
        let packet_send = HashMap::new();
        
        let server = Server::new(
            1,
            controller_send,
            controller_recv,
            packet_recv,
            packet_send,
        );
        
        assert_eq!(server.id, 1);
        assert!(server.packet_send.is_empty());
        assert!(server.topology.is_none());
        assert!(matches!(server.server_type, ServerType::Undefined));
    }
}
